CREATE TABLE Quoi (
    feur TEXT,
    apagnan INTEGER,
    coubeh TEXT,
    CONSTRAINT pk_feur PRIMARY KEY (feur)
);

CREATE TABLE Restaurant (
    resto TEXT,
    capitalnet INTEGER,
    adresse TEXT
    CONSTRAINT pk_resto PRIMARY KEY (resto)
);

CREATE TABLE Poney (
    nom TEXT,
    age INTEGER,
    typemagie TEXT,
    adresse TEXT,
    CONSTRAINT pk_typemagie PRIMARY KEY (typemagie)
);

CREATE TABLE Cia (
    id_agent INTEGER,
    nom_agent TEXT,
    prenom_agent TEXT,
    age_agent TEXT,
    adresse_agent TEXT,
    CONSTRAINT pk_id PRIMARY KEY (id_agent)
);
